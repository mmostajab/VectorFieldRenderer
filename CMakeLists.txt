message ("user define: CMAKE_CXX_FLAGS is ${CMAKE_CXX_FLAGS}")
message ("user define: CMAKE_CXX_FLAGS_RELEASE is ${CMAKE_CXX_FLAGS_RELEASE}")

if(CMAKE_BUILD_TYPE STREQUAL Debug)
	set (BUILD_PREFIX "-d")

        set (BUILD_PRE FIX "")
endif()
